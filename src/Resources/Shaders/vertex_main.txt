#version 450 core 

layout (location = 0) in vec3 vertPos;
layout (location = 1) in vec2 vertTex;
layout (location = 2) in vec3 vertNorm;

out vec2 fragTex;
out vec3 fragPos;
out vec3 fragNorm;

uniform mat4 model;
uniform mat4 view; 
uniform mat4 proj; 

void main() {
	gl_Position = proj * view * model * vec4(vertPos, 1.0);
	fragTex = vec2(vertTex.x, 1.0 - vertTex.y);
	fragPos = (model * vec4(vertPos, 1.0)).xyz;
	fragNorm = mat3(model) * vertNorm; 
}
